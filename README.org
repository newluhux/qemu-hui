** build

#+BEGIN_SRC shell
  git clone https://gitlab.com/qemu-project/qemu /path/to/qemu
  git clone https://github.com/newluhux/qemu-hui/
  cd qemu
  for i in ../qemu-hui/*.patch
  do
      git am $i
  done
#+END_SRC

#+BEGIN_SRC shell
 cd /path/to/qemu/
 ./configure --target-list=arm-softmmu,mipsel-softmmu,riscv64-softmmu
 make -j$(nproc)
 make check
#+END_SRC
